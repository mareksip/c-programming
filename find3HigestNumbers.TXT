#include "stdafx.h"
#include "iostream"
#include <time.h>
using namespace std;

int _tmain(int argc, _TCHAR* argv[])
{
	int arr[1000];
	srand(time(NULL));

	for (int i = 0; i < 1000; i++)
	{
		arr[i] = rand();
		cout << arr[i] << endl;
	}

	for (int c = 0; c<1000; ++c)  
	{
	
		if (arr[0] < arr[c]) 
		{
			int	a = arr[0];
			arr[0] = arr[c];	
			arr[c] = a;
		}
	
	}

	for (int c = 1; c<1000; ++c)  
	{
		if (arr[1] < arr[c]) 
		{
			int a = arr[1];
			arr[1] = arr[c];		
			arr[c] = a;
		}
	}

	for (int c = 2; c<1000; ++c) 
	{
		if (arr[2] < arr[c]) 
		{
			int a = arr[2];
			arr[2] = arr[c];		
			arr[c] = a;
		}
	}

	printf("First largest element = %d\n", arr[0]);
	printf("Second largest element = %d\n", arr[1]);
	printf("Third largest element = %d\n", arr[2]);

	system("pause");
	return 0;
}
